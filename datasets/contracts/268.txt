use ink;

mod {{contract_name}} {
    #[ink(storage)]
    pub struct {{ContractName}} {
        value: bool,
    }

    impl {{ContractName}} {
        #[ink(constructor)]
        pub fn new(init_value: bool) -> Self {
            Self {
                value: init_value
            }
        }

        #[ink(constructor)]
        pub fn default() -> Self {
            Self::new(Default::default())
